const config = {
    data,
    sourceField: 'source',
    targetField: 'target',
    weightField: 'value',
    
    // 节点样式
    nodeStyle: {
        opacity: 1,
        fillOpacity: 1,
    },
    
    // 边样式
    edgeStyle: {
        fillOpacity: 0.4,
    },
    
    // 颜色配置
    color: ['#5B8FF9', '#5AD8A6', '#F6BD16', '#945FB9', '#E86452'],
    
    // 节点宽度
    nodeWidthRatio: 0.01,
    nodePaddingRatio: 0.03,
    
    // 标签配置
    label: {
        formatter: ({ name }) => name,
        style: { fill: '#1D1D1F', fontSize: 11 },
    },
};
桑基图规范
参数值说明
sourceField string 源节点字段
targetField string 目标节点字段
weightField string 流量权重字段
nodeWidthRatio 0.01 节点宽度比例
nodePaddingRatio 0.03 节点间距比例
nodeStyle.fillOpacity 1 节点填充透明度
edgeStyle.fillOpacity 0.4 边填充透明度
label.style.fontSize 11px 标签字号
适用场景
✓ 适合
  • 用户行为路径分析
  • 能源/资金流向展示
  • 系统数据流转
  • 分类归因分析
✗ 不适合
  • 简单占比数据
  • 时间序列趋势
  • 节点过多的场景（>20个）
  • 无明确流向关系的数据
