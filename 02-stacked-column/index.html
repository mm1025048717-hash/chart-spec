Chart Configuration 
const config = {
    data: stackData,
    xField: 'month',
    yField: 'value',
    seriesField: 'type',
    isStack: true,
    padding: [40, 20, 50, 50],
    color: ['#5B8FF9', '#5AD8A6'],
    columnWidthRatio: 0.5,
    // 堆叠图：仅顶部系列有圆角
    columnStyle: (datum) => {
        if (datum.type === '线上') {
            return { radius: [4, 4, 0, 0] };
        }
        return { radius: 0 };
    },
    xAxis: {
        label: { style: { fontSize: 11, fill: '#86868B' } },
        line: { style: { stroke: '#E8E8ED' } },
        tickLine: null,
    },
    yAxis: {
        label: { style: { fontSize: 11, fill: '#86868B' } },
        grid: { line: { style: { stroke: '#E8E8ED', lineDash: [4, 4] } } },
    },
    legend: false,
};
