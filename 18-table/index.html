import { Table, Button, Modal, Input, Space, Tag } from 'antd';
import { FilterOutlined } from '@ant-design/icons';

// 统一筛选组件
const FilterPanel = ({ visible, onClose, onApply, onReset }) => {
    return (
        <Modal
            title="筛选条件"
            open={visible}
            onCancel={onClose}
            footer={[
                <Button key="reset" onClick={onReset}>重置</Button>,
                <Button key="apply" type="primary" onClick={onApply}>应用筛选</Button>,
            ]}
        >
            {/* 所有维度的筛选选项 */}
            <Space direction="vertical" style={{ width: '100%' }}>
                {/* 地区筛选 */}
                <div>
                    <div style={{ fontWeight: 600, marginBottom: 8 }}>地区</div>
                    <Space wrap>
                        {['华东区', '华北区', '华南区', '华中区'].map(region => (
                            <Button key={region} onClick={() => toggleFilter('region', region)}>
                                {region}
                            </Button>
                        ))}
                    </Space>
                </div>
                
                {/* 销售额范围 */}
                <div>
                    <div style={{ fontWeight: 600, marginBottom: 8 }}>销售额</div>
                    <Space>
                        <Input placeholder="最小值" type="number" />
                        <span>至</span>
                        <Input placeholder="最大值" type="number" />
                    </Space>
                </div>
                
                {/* 其他维度... */}
            </Space>
        </Modal>
    );
};

// 表格列配置 - 所有列可点击筛选
const columns = [
    {
        title: '地区',
        dataIndex: 'region',
        key: 'region',
        onHeaderCell: () => ({
            onClick: () => openFilterPanel('region'),
            style: { cursor: 'pointer' }
        }),
        onCell: (record) => ({
            onClick: () => openFilterPanel('region', record.region),
            style: { cursor: 'pointer' }
        }),
    },
    {
        title: '销售额',
        dataIndex: 'sales',
        key: 'sales',
        render: (val) => `¥ ${val.toLocaleString()}`,
        onHeaderCell: () => ({
            onClick: () => openFilterPanel('sales'),
            style: { cursor: 'pointer' }
        }),
        onCell: (record) => ({
            onClick: () => openFilterPanel('sales', record.sales),
            style: { cursor: 'pointer' }
        }),
    },
];

// 使用

表格规范
元素规范说明
表头背景 #FAFAFA 浅灰背景区分表头
表头字号 13px 字重600，颜色 #1D1D1F
单元格内边距 12px 16px 适中的内边距，确保可读性
单元格字号 13px 字重400，颜色 #1D1D1F
边框 #E8E8E8 仅底边框，1px
悬停背景 #F5F5F5 行悬停效果
斑马纹 可选 偶数行背景 #FAFAFA，增强可读性
使用建议
建议说明
列数控制建议不超过 8 列，过多会影响横向阅读
排序功能数值列建议支持排序，提升数据探索能力
分页显示数据量超过 20 条建议使用分页，提升性能
适用场景适合展示结构化数据，如销售明细、用户列表、统计数据等
