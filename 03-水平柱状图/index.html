// 水平柱状图 - Y轴标签智能处理 const SmartYAxis = { getConfig(data, yField, options = {}) { const { maxWidth = 120, truncateAt = 12 } = options; const maxLen = Math.max(...data.map(d => String(d[yField] || '').length)); // 根据最长文字计算Y轴宽度 let axisWidth = 60; if (maxLen > 6 && maxLen <= 10) axisWidth = 90; else if (maxLen > 10 && maxLen <= 15) axisWidth = 100; else if (maxLen > 15) axisWidth = maxWidth; return { label: { formatter: (text) => { const str = String(text || ''); if (str.length > truncateAt) { return str.substring(0, truncateAt) + '...'; } return str; }, style: { fontSize: 11, fill: '#646A73' } }, // 返回计算出的宽度供padding使用 _axisWidth: axisWidth }; } }; // 使用示例 const yAxisConfig = SmartYAxis.getConfig(data, 'product'); new Bar('container', { data, xField: 'value', yField: 'product', // 根据Y轴宽度设置padding padding: [20, 40, 30, yAxisConfig._axisWidth], // 柱子配置 barWidthRatio: 0.6, barStyle: { radius: [0, 4, 4, 0] }, // 右侧圆角 color: '#5B8FF9', // Y轴配置 yAxis: yAxisConfig, // 数据标签 label: { position: 'right', style: { fontSize: 10, fill: '#5B8FF9' } }, // Tooltip显示完整信息 tooltip: { formatter: (datum) => ({ name: datum.product, // 完整显示 value: datum.value.toLocaleString() }) } }).render();
